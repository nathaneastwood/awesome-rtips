2.1 Sample Using a Vector of Sizes

nested_iris <- 
  iris %>% 
  group_by(Species) %>% 
  nest() %>% 
  mutate(n = 2:4)
nested_iris %>% 
  mutate(subsamp = map2(data, n, sample_n)) %>% 
  select(Species, subsamp) %>% 
  unnest()
